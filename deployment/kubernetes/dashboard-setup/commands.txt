
k apply -f dashboard-adminuser.yml
k apply -f dashboard-rolebinding.yml

# 1. for temporary access, create a token and access the dashboard at https://localhost:8443
k -n kubernetes-dashboard create token admin-user

# 2. for long term access,
k apply -f secret.yml
kubectl get secret admin-user -n kubernetes-dashboard -o jsonpath="{.data.token}" | base64 -d

# cleanup
kubectl -n kubernetes-dashboard delete serviceaccount admin-user
kubectl -n kubernetes-dashboard delete clusterrolebinding admin-user



kubectl -n kubernetes-dashboard port-forward svc/kubernetes-dashboard-kong-proxy 8443:443
