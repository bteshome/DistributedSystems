

# install from the remote repo
helm repo add prometheus-community https://prometheus-community.github.io/helm-charts
helm repo update
helm install p prometheus-community/prometheus
k expose service prom-prometheus-server --type=NodePort --target-port=9090 --name=prom-ext
k describe service prom-ext

export POD_NAME=$(kubectl get pods --namespace default -l "app.kubernetes.io/name=prometheus,app.kubernetes.io/instance=p" -o jsonpath="{.items[0].metadata.name}")
kubectl --namespace default port-forward $POD_NAME 9090
